68      /** Join Correct Channels. Set name and topic suppression flags.
69       * @param event Information derived from the event.
70       */
71      @Override
72      public void onConnect(final ConnectEvent<IrcAgent> event) {
73          /*
74           * String webirc =
75           * plugin.getConfig().getString("server.webirc_password"); if(webirc !=
76           * null) { this.sendRawLine(String.format("WEBIRC %s %s %s %s", webirc,
77           * player.getName(), player.getAddress().getHostName(),
78           * player.getAddress().getAddress())); }
79           */
80  
81          // The player may have not gotten then name they wanted.
82          event.getBot().getPlayer().setDisplayName(event.getBot().getNick());
83          event.getBot().getSettings().setIrcNick(event.getBot().getNick());
84  
85          boolean bSuppressNames = plugin.getConfig().getBoolean("suppress.initial_userlist", false);
86          boolean bSuppressTopic = plugin.getConfig().getBoolean("suppress.initial_topic", false);
87          List<?> channelData = plugin.getConfig().getList("default.channels");
88          for (Object i : channelData) {
89              if (i instanceof LinkedHashMap) {
90                  LinkedHashMap<?, ?> linkedHashMapI = (LinkedHashMap<?, ?>) i;
91  
92                  String channelName = (String) linkedHashMapI.get("channel");
93                  if (channelName != null) {
94                      if (bSuppressNames) {
95                          event.getBot().getSuppressNames().add(event.getBot().getChannel(channelName));
96                      }
97                      if (bSuppressTopic) {
98                          event.getBot().getSuppressTopic().add(event.getBot().getChannel(channelName));
99                      }
100                     String key = (String) linkedHashMapI.get("key");
101                     if (key != null) {
102                         event.getBot().joinChannel(channelName, key);
103                     } else {
104                         event.getBot().joinChannel(channelName);
105                     }
106                 }
107             } else {
108                 log.log(Level.WARNING, "Object: {0} is a {1}", new Object[]{i.toString(), i.getClass().toString()});
109             }
110         }
111     }

157     /**
158      * Message received handler.
159      * @param event Information derived from the event.
160      */
161     @Override
162     public void onMessage(final MessageEvent<IrcAgent> event) {
163         String formattedMessage = plugin.getConfig().getString("messages.chat-irc");
164         String channel = event.getChannel().getName();
165         String sender = event.getUser().getNick();
166 
167         try {
168             int playerId = Integer.parseInt(event.getUser().getLogin().substring(1));
169             IrcAgent agent = plugin.getBots().get(playerId);
170             if (null != agent) {
171                 Player player = agent.getPlayer();
172                 if (null != player && sender.equals(player.getDisplayName())) {
173                     sender = player.getDisplayName();
174                 }
175             }
176         } catch (NumberFormatException ex) {
177             sender = event.getUser().getNick();
178         }
179 
180         String message = event.getMessage();
181         formattedMessage = formattedMessage.replace("${CHANNEL}", channel);
182         formattedMessage = formattedMessage.replace("${NICK}", sender);
183         formattedMessage = formattedMessage.replace("${MESSAGE}", message);
184         event.getBot().getPlayer().sendMessage(formattedMessage.replace("&", "\u00A7"));
185         log.info(formattedMessage);
186     }

196         if (event.getBot().getUserBot().getNick().equals(event.getOldNick())) {
197             event.getBot().getPlayer().setDisplayName(event.getNewNick());
198             event.getBot().getSettings().setIrcNick(event.getNewNick());
199             event.getBot().saveSettings();
200         }
201         String format = plugin.getConfig().getString("messages.rename");
202         String oldNick = event.getOldNick();
203         String newNick = event.getNewNick();
204         String message = format.replace("${OLDNICK}", oldNick).replace("${NEWNICK}", newNick);
205         event.getBot().getPlayer().sendMessage(message.replace("&", "\u00A7"));

248     /**
249      * Handles response codes not handled by pircbot This methods handles irc
250      * response codes, slices up the response, and then calls the appropriate
251      * method.
252      * @param event Information derived from the event.
253      */
254     @Override
255     public void onUnknown(final UnknownEvent<IrcAgent> event) {
256         Pattern responsePattern = Pattern.compile("(\\S*) (\\S*) :(.*)");
257         Matcher responseMatcher = responsePattern.matcher(event.getLine());
258         responseMatcher.find();
259         int code = Integer.parseInt(responseMatcher.group(1));
260         switch (code) {
261             case ReplyConstants.ERR_NOSUCHNICK: // TODO this needs a clearer error message
262             case ReplyConstants.ERR_NOSUCHCHANNEL:
263             case ReplyConstants.ERR_NICKNAMEINUSE:
264             case ReplyConstants.ERR_INVITEONLYCHAN:
265             case ReplyConstants.ERR_BADCHANNELKEY:
266                 onErrorMessage(event, responseMatcher.group(2), responseMatcher.group(3));  //NOPMD
267                 break;
268             default:
269                 break;
270 
271         }
272     }

286     /**
287      * Handles topic responses. Topic responses come in: a: as a response to a
288      * topic request b: on channel join c: on topic change Honors muteTopic flag
289      * for channel, will clear it as well.
290      * @param event Information derived from the event.
291      */
292     @Override
293     public void onTopic(final TopicEvent<IrcAgent> event) {
294         String format;
295         String channel = event.getChannel().getName();
296         String topic = event.getChannel().getTopic();
297         Player player = event.getBot().getPlayer();
298 
299         if (event.getBot().getSuppressTopic().contains(event.getChannel())) {
300             event.getBot().getSuppressTopic().remove(event.getChannel());
301         } else {
302             format = plugin.getConfig().getString("messages.topic");
303             String formattedMessage = format.replace("${CHANNEL}", channel).replace("${TOPIC}", topic);
304             player.sendMessage(formattedMessage.replace("&", "\u00A7"));
305         }
306     }

308     /**
309      * UserList Response handler. usually a response to /names. Also occurs on
310      * channel join. This will check the muteNames set, and clear it.
311      * @param event Information derived from the event.
312      */
313     @Override
314     public void onUserList(final UserListEvent<IrcAgent> event) {
315         if (event.getBot().getSuppressNames().contains(event.getChannel())) {
316             event.getBot().getSuppressNames().remove(event.getChannel());
317         } else {
318             StringBuilder usersString = new StringBuilder();
319             for (User user : event.getUsers()) {
320                 usersString.append(user.getNick());
321                 usersString.append(" ");
322             }
323             String format = plugin.getConfig().getString("messages.list");
324             String channel = event.getChannel().getName();
325             String message = format.replace("${LIST}", usersString.toString()).replace("${CHANNEL}", channel);
326             event.getBot().getPlayer().sendMessage(message.replace("&", "\u00A7"));
327         }
