39      /**
40       * Convert a MineCraft color code to an IRC color code
41       * 
42       * @param code
43       *            A MineCraft color code
44       * @return An IRC color code, black if the string did not match any code.
45       */
46      public static String chatToIrcColor(String code) {
47          for (int i = 0; i < minecraftColor.size(); i++) {
48              if (minecraftColor.get(i).equals(code)) {
49                  String result = "\u0003";
50                  if (i < 10)
51                      result += "0";
52                  return result + i;
53              }
54          }
55          return "\u000301"; // Let's just make it black by default, good for IRC
56      }

58      /**
59       * Convert message from IRC to Minecraft Translates a colored irc message
60       * into a colored minecraft message.
61       * 
62       * @param message
63       *            the incoming irc message.
64       * @return The converted minecraft string. TODO minecraft color doesn't span
65       *         line wrap, force it by checking for last active color at
66       *         linebreak.
67       */
68      public static String fromIrc(String message) {
69          // define IRC color pattern
70          Matcher m = ircColorPattern.matcher(message);
71          String result = "";
72          int prev = 0;
73          // Find all matches
74          while (m.find()) {
75              int start = m.start();
76              int end = m.end();
77              // Add the unmatched parts to the result
78              result += message.substring(prev, start);
79              char digit1 = message.charAt(start + 1);
80              char digit2 = 'a';
81              if (start + 2 < message.length())
82                  digit2 = message.charAt(start + 2);
83              // We don't need to catch a numberformatexception
84              // because the regular expression took care of that
85              String color = "" + digit1
86                      + (Character.isDigit(digit2) ? digit2 : "");
87              int code = Integer.parseInt(color);
88              // Replace matched parts by the other color code
89              result += minecraftColor.get(code);
90              prev = end;
91          }
92          // Add the remaining string
93          if (prev < message.length())
94              result += message.substring(prev, message.length());
95  
96          return result;
97      }
