76      /**
77       * Convert message from IRC to Minecraft Translates a colored irc message
78       * into a colored minecraft message.
79       * @param message
80       *            the incoming irc message.
81       * @return The converted minecraft string. TODO minecraft color doesn't span
82       *         line wrap, force it by checking for last active color at
83       *         linebreak.
84       */
85      public static String fromIrc(final String message) {
86          // define IRC color pattern
87          Matcher m = ircColorPattern.matcher(message);
88          StringBuilder result = new StringBuilder();
89          int prev = 0;
90          // Find all matches
91          while (m.find()) {
92              int start = m.start();
93              int end = m.end();
94              // Add the unmatched parts to the result
95              result.append(message.substring(prev, start));
96              char digit1 = message.charAt(start + 1);
97              char digit2 = 'a';
98              if (start + 2 < message.length()) {
99                  digit2 = message.charAt(start + 2);
100             }
101             // We don't need to catch a numberformatexception
102             // because the regular expression took care of that
103             String color;
104             if (Character.isDigit(digit2)) {
105                 color = "" + '\u0003' + digit1 + digit2;
106             } else {
107                 color = "" + '\u0003' + "0" + digit1;
108             }
109             // Replace matched parts by the other color code
110             result.append(ircToMinecraftColor.get(color));
111             prev = end;
112         }
113         // Add the remaining string
114         if (prev < message.length()) {
115             result.append(message.substring(prev, message.length()));
116         }
117 
118         return result.toString();
119     }
