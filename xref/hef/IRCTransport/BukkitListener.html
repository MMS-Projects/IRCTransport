14  /**
15   * Handle events for all Player related events.

17  public class BukkitListener implements Listener {
18  
19      /** Maps to retrieve associated IrcAggent from player. */
20      private ConcurrentHashMap<Integer, IrcAgent> bots;
21      /** Reference to the parent plugin. */
22      private final IRCTransport plugin;
23      /** Logger object. */
24      private Logger log;

26      /**
27       * @param instance A reference to the plugin.
28       */
29      public BukkitListener(final IRCTransport instance) {
30          this.bots = instance.getBots();
31          plugin = instance;
32          log = instance.getServer().getLogger();
33      }
34  
35      /**
36       * Player chat handler.  It cancels that chat event.
37       * @param event the player chat event.
38       */
39      @EventHandler
40      public void onPlayerChat(final PlayerChatEvent event) {
41          IrcAgent bot = this.bots.get(event.getPlayer().getEntityId());
42          if (null != bot && bot.isConnected()) {
43              bot.sendMessage(event.getMessage());
44              // prevent messages from being displayed twice.
45              event.setCancelled(true);
46          }
47      }
48  
49      /**
50       * Creates Player related objects.
51       * @param event player join event.
52       */
53      @EventHandler
54      public void onPlayerJoin(final PlayerJoinEvent event) {
55          Player player = event.getPlayer();
56          InitializeAgent initializeAgent = new InitializeAgent(plugin, player);
57          plugin.getServer().getScheduler().scheduleAsyncDelayedTask(plugin, initializeAgent);
58      }
59  
60      /**
61       * Removes player from list of players.
62       * @param event the player quit event.
63       */
64      @EventHandler
65      public void onPlayerQuit(final PlayerQuitEvent event) {
66          Player player = event.getPlayer();
67          int playerID = player.getEntityId();
68          IrcAgent agent = bots.get(playerID);
69          if (null != agent) {
70              agent.shutdown();
71              bots.remove(playerID);
72              log.info(String.format("Removed agent for Player ID: %d name: %s", playerID, player.getName()));
73          }
74      }
75  
76      /**
77       * Sends death message as action on player death.
78       * @param event PlayerDeathEvent
79       */
80      @EventHandler
81      public void onPlayerDeath(final PlayerDeathEvent event) {
82          String deathMessage = event.getDeathMessage();
83  
84          IrcAgent bot = this.bots.get(event.getEntity().getEntityId());
85          if (null != bot && bot.isConnected() && deathMessage != null && !deathMessage.equals("")) {
86              String playerName = ((Player) event.getEntity()).getName();
87  
88              // deathMessage seems to be 'playerName died for some reason'
89              if (deathMessage.startsWith(playerName)) {
90                  // So, we just remove the leading playerName, so deaths render as:
91                  // * ircNick died for some reason
92                  deathMessage = deathMessage.substring(playerName.length());
93              } else {
94                  // Otherwise, just try to make it look reasonable
95                  deathMessage = "died: " + deathMessage;
96              }
97              bot.sendAction(deathMessage.trim());
98              // This winds up being sent to the player via IRC,
99              // so suppress the default message
100             event.setDeathMessage(null);
101         }
102     }
