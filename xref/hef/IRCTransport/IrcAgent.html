57  				player.getName()));
58  		if (null == getSettings()) {
59  			setSettings(new AgentSettings(player));
60  			String prefix = plugin.getConfig().getString("default.prefix");
61  			String suffix = plugin.getConfig().getString("default.suffix");
62  			getSettings().setIrcNick(
63  					String.format("%s%s%s", prefix, player.getName(), suffix));
64  		} else {
65  			LOG.log(Level.INFO, String.format(
66  					"Player '%s' using persistent IRC nick '%s'",
67  					player.getName(), getSettings().getIrcNick()));
68  		}
69  		setNick(getSettings().getIrcNick());
70  		new Connect(this).run();
71  	}

168 	@Override
169 	public void onConnect() {
170 		/*
171 		 * String webirc =
172 		 * plugin.getConfig().getString("server.webirc_password"); if(webirc !=
173 		 * null) { this.sendRawLine(String.format("WEBIRC %s %s %s %s", webirc,
174 		 * player.getName(), player.getAddress().getHostName(),
175 		 * player.getAddress().getAddress())); }
176 		 */
177 
178 		// The player may have not gotten then name they wanted.
179 		getPlayer().setDisplayName(getNick());
180 		getSettings().setIrcNick(getNick());
181 
182 		boolean bSuppressNames = plugin.getConfig().getBoolean("suppress.initial_userlist", false);
183 		boolean bSuppressTopic = plugin.getConfig().getBoolean("suppress.initial_topic", false);
184 		List<?> channelData = plugin.getConfig().getList("default.channels");
185 
186 		for (Object i : channelData) {
187 			if (i instanceof LinkedHashMap) {
188 				LinkedHashMap<?, ?> lhm_i = (LinkedHashMap<?, ?>) i;
189 
190 				String channel = (String) lhm_i.get("channel");
191 				if (channel != null) {
192 					if (bSuppressNames) {
193 						suppressNames.add(channel);
194 					}
195 					if (bSuppressTopic) {
196 						suppressTopic.add(channel);
197 					}
198 					String key = (String) lhm_i.get("key");
199 					if (key != null) {
200 						joinChannel(channel, key);
201 					} else {
202 						joinChannel(channel);
203 					}
204 				}
205 			} else {
206 				log("Object: " + i.toString() + " is a "
207 						+ i.getClass().toString());
208 			}
209 		}
210 	}

397 	 * Handles response codes not handled by pircbot This methods handles irc
398 	 * response codes, slices up the response, and then calls the appropriate
399 	 * method.
400 	 * 
401 	 * @param code
402 	 *            the irc response code.
403 	 * @param response
404 	 *            The message that came with the response
405 	 */
406 	@Override
407 	protected void onServerResponse(final int code, final String response) {
408 		Pattern responsePattern = Pattern.compile("\\S* (\\S*) :(.*)");
409 		Matcher responseMatcher = responsePattern.matcher(response);
410 		responseMatcher.find();
411 		switch (code) {
412 		case ERR_NOSUCHNICK: // TODO this needs a clearer error message
413 		case ERR_NOSUCHCHANNEL:
414 		case ERR_NICKNAMEINUSE:
415 		case ERR_INVITEONLYCHAN:
416 		case ERR_BADCHANNELKEY:
417 			onErrorMessage(responseMatcher.group(1), responseMatcher.group(2));
418 			break;
419 		default:
420 			break;
421 
422 		}
423 	}

429 	 * @param channel

440 	@Override
441 	protected void onTopic(final String channel, final String topic,
442 			final String setBy, final long date, final boolean changed) {
443 		if (changed) {
444 			getPlayer().sendMessage(
445 					ChatColor.YELLOW
446 							+ String.format("[%s] Topic changed: %s", channel,
447 									topic));
448 		} else {
449 			if (!suppressTopic.contains(channel)) {
450 				getPlayer().sendMessage(
451 						ChatColor.YELLOW
452 								+ String.format("[%s] Topic: %s", channel,

454 			} else {
455 				// Only silence the initial topic response.
456 				suppressTopic.remove(channel);
457 			}
458 		}
