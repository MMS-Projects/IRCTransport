45       * Agent Constructor.
46       * @param instance Reference to plugin instance.
47       * @param bukkitPlayer Reference to Bukkit Player
48       */
49      public IrcAgent(final IRCTransport instance, final Player bukkitPlayer) {
50          this.plugin = instance;
51          this.player = bukkitPlayer;
52          this.shuttingDown = false;
53          setLogin(String.format("%s", player.getEntityId()));
54          super.setAutoNickChange(true);
55  
56          // init player settings
57          setSettings(plugin.getDatabase().find(AgentSettings.class, player.getName()));
58          if (null == getSettings()) {
59              setSettings(new AgentSettings(player));
60              String prefix = plugin.getConfig().getString("default.prefix", "");
61              String suffix = plugin.getConfig().getString("default.suffix", "");
62              getSettings().setIrcNick(String.format("%s%s%s", prefix, player.getName(), suffix));
63          } else {
64              String format = "Player '%s' using persistent IRC nick '%s'";
65              String name = player.getName();
66              String nick = getSettings().getIrcNick();
67              LOG.log(Level.INFO, String.format(format, name, nick));
68          }
69          setNick(getSettings().getIrcNick());
70          new Connect(this).run();
71      }

73      /**
74       * Connect the agent. Don't call this directly, call `new
75       * Connect(this).run()` instead.
76       * @throws IOException If it was not possible to connect to the server.
77       * @throws IrcException If the server would not let us join it.
78       */
79      public void connect() throws IOException, IrcException {
80          String address = getPlugin().getConfig().getString("server.address");
81          int port = getPlugin().getConfig().getInt("server.port");
82          String password = getPlugin().getConfig().getString("server.password");
83  
84          SocketFactory socketFactory = null;
85          if (getPlugin().getConfig().getBoolean("server.ssl", false)) {
86              socketFactory = new UtilSSLSocketFactory();
87          }
88  
89          //setup WEBIRC
90          setWebIrcAddress(this.getPlayer().getAddress().getAddress());
91          setWebIrcHostname(player.getAddress().getHostName());
92          String webIrcPassword = getPlugin().getConfig().getString("server.webirc_password");
93          if (webIrcPassword != null) {
94              this.setWebIrcPassword(webIrcPassword);
95          }
96  
97          if (!isConnected()) {
98              if (getServer() == null) {
99                  connect(address, port, password, socketFactory);
100             } else {
101                 reconnect();
102             }
103         }
104     }
