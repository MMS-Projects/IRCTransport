12   * Handle events for all Player related events.

14   public final class IRCTransportPlayerListener extends PlayerListener {
15      /** Maps to retrieve associated IrcAggent from player. */
16      private HashMap<Player, IrcAgent> bots;
17      /** Reference to the parent plugin. */
18      private final IRCTransport plugin;
19  
20      /**
21       * @param instance A reference to the plugin.
22       */
23      public IRCTransportPlayerListener(final IRCTransport instance) {
24          this.bots = instance.getBots();
25          plugin = instance;
26      }
27  
28      @Override
29      public void onPlayerChat(final PlayerChatEvent event) {
30          IrcAgent bot = this.bots.get(event.getPlayer());
31          if (bot.isConnected()) {
32              bot.sendMessage(event.getMessage());
33              // prevent messages from being displayed twice.
34              event.setCancelled(true);
35          }
36      }
37  
38      @Override
39      public void onPlayerJoin(final PlayerJoinEvent event) {
40          this.bots.put(event.getPlayer(),
41                  new IrcAgent(plugin, event.getPlayer()));
42      }
43  
44      @Override
45      public void onPlayerQuit(final PlayerQuitEvent event) {
46          this.bots.get(event.getPlayer()).shutdown();
47          this.bots.remove(event.getPlayer());
48      }
