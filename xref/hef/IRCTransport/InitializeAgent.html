5   package hef.IRCTransport;

7   import org.bukkit.entity.Player;
8   
9   /**
10   * Initializes an IRC Agent.  This was moved into a Runnable as a result of
11   * profiling.  IRC Agent construction was causing bogs in the main thread.
12   */
13  public class InitializeAgent implements Runnable {
14      /** Handle on plugin. */
15      private IRCTransport plugin;
16      /** Handle on player. */
17      private Player player;
18  
19      /**
20       * Setup the agent initialization job.
21       * @param myPlugin the IRCTRansport plugin
22       * @param myPlayer the bukkit player handle.
23       */
24      InitializeAgent(final IRCTransport myPlugin, final Player myPlayer) {
25          this.plugin = myPlugin;
26          this.player = myPlayer;
27      }
28  
29      /**
30       * Constructs the irc agent,
31       * Adds the agent to the list of bots,
32       * and connects the agent to IRC.
33       */
34      @Override
35      public void run() {
36          int playerID = player.getEntityId();
37          IrcAgent agent = new IrcAgent(plugin, player);
38          agent.getListenerManager().addListener(plugin.getListener());
39          plugin.getBots().put(playerID, agent);
40          plugin.getLogger().info(String.format("Created agent for Player ID: %d name: %s", playerID, player.getName()));
41          new Connect(agent).run();
42      }
