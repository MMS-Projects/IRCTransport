11  /**
12   * Connection Task This class us a Runnable class for connecting to the irc
13   * server. It will reschedule a connection
14   */
15  public final class Connect implements Runnable {
16      /**
17       * Time between connection retries after a failed connection attempt.
18       * 4oo is 20 seconds.
19       */
20      public static final int RETRY_RATE = 400;
21      /**
22       * Amount of time to wait to reconnect if a connection was already
23       * established, but was then lost.
24       * 100 is 5 seconds
25       */
26      public static final int INITIAL_RETRY_DELAY = 100;
27      /** Log object. */
28      private static final Logger LOG = Logger.getLogger("Minecraft");
29      /**
30       * Parent IRC agent. The class will perform a lot of actions on the agent.
31       */
32      private IrcAgent agent;
33  
34      /**
35       * Create a Connection instance Pass this from IrcAgent so we have access.
36       * @param parent
37       *            pass in this from the calling IrcAgent
38       */
39      public Connect(final IrcAgent parent) {
40          this.agent = parent;
41      }
42  
43      /**
44       * Run the connection task This task will schedule another connection task
45       * if it fails to connect.
46       */
47      @Override
48      public void run() {
49          // If a task where scheduled, and the user disconnected, this would
50          // continually keep the agent alive.
51          if (!agent.isShuttingDown()) {
52              try {
53                  // If we never set the server i.e. havn't connected yet
54                  if (agent.getServer() == null) {
55                      //TODO: these settings should be moved into the agent.
56                      agent.connect(agent.getPlugin().getConfig().getString("server"),
57                                    agent.getPlugin().getConfig().getInt("port"),
58                                    agent.getPlugin().getConfig().getString("password"));
59                  } else {
60                      // reconnect should recycle settings the user already has
61                      agent.reconnect();
62                  }
63              } catch (NickAlreadyInUseException e) {
64                  // This should not happen, the agent is set to auto retry new
65                  // names
66                  LOG.log(Level.SEVERE, e.getMessage(), e);
67              } catch (IOException e) {
68                  if (e.getMessage().equalsIgnoreCase("Connection refused")) {
69                      agent.getPlayer().sendMessage(
70                              ChatColor.YELLOW + "Failed to connect to Chat Server.");
71                      // 400 seems to be 20 seconds
72                      agent.getPlugin()
73                              .getServer()
74                              .getScheduler()
75                              .scheduleAsyncDelayedTask(agent.getPlugin(), this, RETRY_RATE);
76                  } else if
77                      (e.getMessage().equalsIgnoreCase("Connection reset")) {
78                      agent.getPlayer().sendMessage(ChatColor.YELLOW + "Connection reset while connecting to Chat Server");
79                      agent.getPlugin()
80                              .getServer()
81                              .getScheduler()
82                              .scheduleAsyncDelayedTask(agent.getPlugin(), this, INITIAL_RETRY_DELAY);
83                  } else {
84                      LOG.log(Level.SEVERE, e.getMessage(), e);
85                  }
86              } catch (IrcException e) {
87                  LOG.log(Level.SEVERE, e.getMessage(), e);
88              }
89          }
90      }
91  }
