11  /** Connection Task
12   * This class us a Runnable class for connecting to the irc server.  It will reschedule a connection
13   * @author hef
14   *
15   */
16  public class Connect implements Runnable {
17  	private IrcAgent agent;
18  	private static final Logger log = Logger.getLogger("Minecraft");
19  	/** Create a Connection instance
20  	 * Pass this from IrcAgent so we have access
21  	 * @param agent pass in this from the calling IrcAgent
22  	 */
23  	public Connect(IrcAgent agent)
24  	{
25  		this.agent=agent;
26  	}
27  	/** Run the connection task
28  	 * This task will schedule another connection task if it fails to connect.
29  	 */
30  	public void run()
31  	{
32  		//If a task where scheduled, and the user disconnected, this would
33  		//continually keep the agent alive.
34  		if(!agent.isShuttingDown())
35  		{
36  			try {
37  				//If we never set the server i.e. havn't connected yet
38  				if(agent.getServer()==null)
39  				{
40  					agent.connect(agent.getPlugin().getIrcServer(), agent.getPlugin().getIrcPort(), agent.getPlugin().getIrcPassword());
41  				}
42  				else
43  				{
44  					//reconnect should recycle settings the user already has
45  					agent.reconnect();
46  				}
47  			} catch (NickAlreadyInUseException e) {
48  				//This should not happen, the agent is set to auto retry new names
49  				log.log(Level.SEVERE, e.getMessage(), e);
50  			} catch (IOException e) {
51  				if(e.getMessage().equalsIgnoreCase("Connection refused"))
52  				{
53  					agent.getPlayer().sendMessage(ChatColor.YELLOW + "Failed to connect to Chat Server.");
54  					//400 seems to be 20 seconds
55  					agent.getPlugin().getServer().getScheduler().scheduleAsyncDelayedTask(agent.getPlugin(), this, 400);
56  				}
57  				else if(e.getMessage().equalsIgnoreCase("Connection reset"))
58  				{
59  					agent.getPlayer().sendMessage(ChatColor.YELLOW + "Connection reset while connecting to Chat Server");
60  					agent.getPlugin().getServer().getScheduler().scheduleAsyncDelayedTask(agent.getPlugin(), this, 100);
61  				}
62  				else
63  				{
64  					log.log(Level.SEVERE, e.getMessage(), e);
65  				}
66  			} catch (IrcException e) {
67  				log.log(Level.SEVERE, e.getMessage(), e);
68  			}
69  
70  			//The player may have not gotten then name they wanted.
71  			agent.getPlayer().setDisplayName(agent.getNick());
72  			agent.getSettings().setIrcNick(agent.getNick());
73  			if(!agent.getPlugin().getAutoJoin().equals(""))
74  			{
75  				//if no channel key is set
76  				if(agent.getPlugin().getAutoJoinKey().equals(""))
77  				{
78  					agent.joinChannel(agent.getPlugin().getAutoJoin());
79  				}
80  				else // channel key is set
81  				{
82  					agent.joinChannel(agent.getPlugin().getAutoJoin(),agent.getPlugin().getAutoJoinKey());
83  				}
84  				agent.setActiveChannel(agent.getPlugin().getAutoJoin());
85  			}
86  		}
87  	}
